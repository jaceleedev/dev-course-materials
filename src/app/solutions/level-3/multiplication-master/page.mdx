# 🔢 구구단 마스터 (답안)

## 🚀 기본 기능 답안

```python
# 🔢 구구단 마스터 - 기본 기능
print("🔢 구구단 마스터에 오신 것을 환영합니다!")
print("="*40)

# 단수 입력 및 검증
while True:
    dan = int(input("몇 단을 출력할까요? (2-9): "))
    if 2 <= dan <= 9:
        break
    else:
        print("❌ 2~9 사이의 숫자를 입력하세요!")

print(f"\n📚 {dan}단")
print("-"*15)

# 구구단 출력
for i in range(1, 10):
    result = dan * i
    print(f"{dan} × {i} = {result}")

print("="*40)
print("🎉 구구단 출력 완료!")
```

## 🎨 추가 기능 1: 구구단 퀴즈 게임

```python
# 🔢 구구단 퀴즈 게임
import random

print("🎮 구구단 퀴즈 게임에 오신 것을 환영합니다!")
print("="*50)

score = 0
total_questions = 5

print("📝 5문제를 출제합니다!")
print("-"*30)

for question_num in range(1, total_questions + 1):
    # 랜덤 문제 생성
    dan = random.randint(2, 9)  # 2~9단 중 랜덤
    num = random.randint(1, 9)  # 1~9 중 랜덤
    answer = dan * num

    print(f"\n📝 문제 {question_num}: {dan} × {num} = ?")
    user_answer = int(input("답: "))

    if user_answer == answer:
        print("🎉 정답입니다!")
        score += 1
    else:
        print(f"❌ 틀렸습니다. 정답은 {answer}입니다.")

# 최종 결과
print(f"\n📊 최종 점수: {score}/{total_questions}")

# 성적 평가
if score == total_questions:
    print("🏆 완벽해요! 구구단 마스터!")
elif score >= 4:
    print("🌟 우수해요!")
elif score >= 3:
    print("👍 잘했어요!")
else:
    print("📚 조금 더 연습해보세요!")

print("="*50)
```

## 💡 추가 기능 2: 구구단 표 만들기

```python
# 🔢 구구단 표 만들기
print("📚 구구단 표 만들기")
print("="*30)

# 범위 입력
while True:
    start_dan = int(input("시작 단 (2-9): "))
    if 2 <= start_dan <= 9:
        break
    else:
        print("❌ 2~9 사이의 숫자를 입력하세요!")

while True:
    end_dan = int(input("끝 단 (2-9): "))
    if start_dan <= end_dan <= 9:
        break
    else:
        print(f"❌ {start_dan}~9 사이의 숫자를 입력하세요!")

print(f"\n📊 {start_dan}단부터 {end_dan}단까지")
print("="*50)

# 여러 단 출력
for dan in range(start_dan, end_dan + 1):
    print(f"\n=== {dan}단 ===")
    print("-"*15)

    for i in range(1, 10):
        result = dan * i
        print(f"{dan} × {i} = {result}")

print("\n🎉 구구단 표 출력 완료!")
```

## 🔥 추가 기능 3: 종합 메뉴 시스템

```python
# 🔢 구구단 마스터 - 종합 시스템
import random

print("🔢 구구단 마스터에 오신 것을 환영합니다!")
print("="*50)

while True:
    print("\n🎮 메뉴를 선택하세요:")
    print("="*30)
    print("1. 원하는 단 출력")
    print("2. 구구단 퀴즈 게임")
    print("3. 구구단 표 만들기")
    print("4. 종료")
    print("="*30)

    choice = input("선택하세요 (1-4): ")

    if choice == "1":
        # 기본 기능 - 원하는 단 출력
        print("\n📚 원하는 단 출력")

        while True:
            dan = int(input("몇 단을 출력할까요? (2-9): "))
            if 2 <= dan <= 9:
                break
            else:
                print("❌ 2~9 사이의 숫자를 입력하세요!")

        print(f"\n📚 {dan}단")
        print("-"*15)

        for i in range(1, 10):
            result = dan * i
            print(f"{dan} × {i} = {result}")

    elif choice == "2":
        # 추가 기능 1 - 구구단 퀴즈 게임
        print("\n🎮 구구단 퀴즈 게임!")

        score = 0
        total_questions = 5

        print("📝 5문제를 출제합니다!")
        print("-"*30)

        for question_num in range(1, total_questions + 1):
            dan = random.randint(2, 9)
            num = random.randint(1, 9)
            answer = dan * num

            print(f"\n📝 문제 {question_num}: {dan} × {num} = ?")
            user_answer = int(input("답: "))

            if user_answer == answer:
                print("🎉 정답입니다!")
                score += 1
            else:
                print(f"❌ 틀렸습니다. 정답은 {answer}입니다.")

        print(f"\n📊 최종 점수: {score}/{total_questions}")

        if score == total_questions:
            print("🏆 완벽해요! 구구단 마스터!")
        elif score >= 4:
            print("🌟 우수해요!")
        elif score >= 3:
            print("👍 잘했어요!")
        else:
            print("📚 조금 더 연습해보세요!")

    elif choice == "3":
        # 추가 기능 2 - 구구단 표 만들기
        print("\n📚 구구단 표 만들기")

        while True:
            start_dan = int(input("시작 단 (2-9): "))
            if 2 <= start_dan <= 9:
                break
            else:
                print("❌ 2~9 사이의 숫자를 입력하세요!")

        while True:
            end_dan = int(input("끝 단 (2-9): "))
            if start_dan <= end_dan <= 9:
                break
            else:
                print(f"❌ {start_dan}~9 사이의 숫자를 입력하세요!")

        print(f"\n📊 {start_dan}단부터 {end_dan}단까지")
        print("="*50)

        for dan in range(start_dan, end_dan + 1):
            print(f"\n=== {dan}단 ===")
            print("-"*15)

            for i in range(1, 10):
                result = dan * i
                print(f"{dan} × {i} = {result}")

        print("\n🎉 구구단 표 출력 완료!")

    elif choice == "4":
        print("\n👋 구구단 마스터를 이용해주셔서 감사합니다!")
        break

    else:
        print("❌ 1~4 중에서 선택하세요!")

print("🎉 프로그램을 종료합니다!")
```

## 💡 핵심 포인트

1. **기본 문법**: 변수, 조건문, 반복문, 입출력만 사용
2. **입력 검증**: while문으로 올바른 입력까지 반복
3. **랜덤 사용**: `import random`, `random.randint()` 활용
4. **메뉴 시스템**: while True와 if-elif로 반복 메뉴 구현
5. **단계적 확장**: 기본 → 퀴즈 → 표 → 메뉴 순서로 기능 확장
