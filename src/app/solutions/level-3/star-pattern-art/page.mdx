# ⭐ 별찍기 아트 (답안)

## 🚀 기본 기능 답안

```python
# ⭐ 별찍기 아트 - 기본 기능
print("⭐ 별찍기 아트에 오신 것을 환영합니다!")
print("="*40)

# 패턴 높이 입력
while True:
    height = int(input("패턴 크기 (3-15): "))
    if 3 <= height <= 15:
        break
    else:
        print("❌ 3~15 사이의 숫자를 입력하세요!")

print(f"\n📐 {height}줄 패턴들을 출력합니다!")
print("="*40)

# 1. 왼쪽 정렬 삼각형
print("⭐ 왼쪽 정렬 삼각형")
print("-" * 20)

for i in range(1, height + 1):
    for j in range(i):
        print("*", end="")
    print()  # 줄바꿈

print()

# 2. 오른쪽 정렬 삼각형
print("⭐ 오른쪽 정렬 삼각형")
print("-" * 20)

for i in range(1, height + 1):
    # 공백 출력
    for j in range(height - i):
        print(" ", end="")

    # 별 출력
    for j in range(i):
        print("*", end="")

    print()  # 줄바꿈

print()

# 3. 이등변삼각형
print("⭐ 이등변삼각형")
print("-" * 20)

for i in range(1, height + 1):
    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 별 출력 (홀수 개씩)
    for j in range(2 * i - 1):
        print("*", end="")

    print()  # 줄바꿈

print()
print("🎉 기본 패턴 출력 완료!")
```

## 🎨 추가 기능 1: 고급 패턴

```python
# ⭐ 고급 별찍기 패턴
print("⭐ 고급 별찍기 패턴")
print("="*30)

# 패턴 높이 입력
while True:
    height = int(input("패턴 크기 (3-15): "))
    if 3 <= height <= 15:
        break
    else:
        print("❌ 3~15 사이의 숫자를 입력하세요!")

print(f"\n📐 {height}줄 고급 패턴들:")
print("="*30)

# 1. 다이아몬드 패턴
print("💎 다이아몬드 패턴")
print("-" * 20)

# 위쪽 삼각형 (중앙 포함)
for i in range(1, height + 1):
    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 별 출력
    for j in range(2 * i - 1):
        print("*", end="")

    print()

# 아래쪽 삼각형
for i in range(height - 1, 0, -1):
    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 별 출력
    for j in range(2 * i - 1):
        print("*", end="")

    print()

print()

# 2. 역삼각형
print("🔻 역삼각형")
print("-" * 20)

for i in range(height, 0, -1):  # height부터 1까지 감소
    # 앞쪽 공백 (점점 증가)
    for j in range(height - i):
        print(" ", end="")

    # 별 출력 (점점 감소)
    for j in range(2 * i - 1):
        print("*", end="")

    print()

print()

# 3. 속이 빈 삼각형
print("🔺 속이 빈 삼각형")
print("-" * 20)

for i in range(1, height + 1):
    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 각 줄별 처리
    for j in range(2 * i - 1):
        if i == 1 or i == height:  # 첫 줄 또는 마지막 줄
            print("*", end="")
        else:  # 중간 줄들
            if j == 0 or j == 2 * i - 2:  # 양끝만
                print("*", end="")
            else:  # 가운데는 공백
                print(" ", end="")

    print()

print()

# 4. 피라미드 패턴
print("🏛️ 피라미드 패턴")
print("-" * 20)

for i in range(1, height + 1):
    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 별과 공백을 번갈아가며
    for j in range(2 * i - 1):
        if j % 2 == 0:  # 홀수 번째 위치에만 별
            print("*", end="")
        else:
            print(" ", end="")

    print()

print()
print("🎉 고급 패턴 출력 완료!")
```

## 💡 추가 기능 2: 커스터마이징 시스템

```python
# ⭐ 커스터마이징 시스템 (기본 변수와 반복문만 사용)
print("⭐ 사용자 정의 패턴 생성기")
print("="*35)

# 패턴 크기 입력
while True:
    size = int(input("패턴 크기 (3-20): "))
    if 3 <= size <= 20:
        break
    else:
        print("❌ 3~20 사이의 숫자를 입력하세요!")

# 사용할 문자 입력
char = input("사용할 문자를 입력하세요 (기본값: *): ")
if not char:
    char = "*"

# 패턴 종류 선택
print("\n🎨 패턴 종류를 선택하세요:")
print("1. 이등변삼각형")
print("2. 다이아몬드")
print("3. 역삼각형")
print("4. 속이 빈 삼각형")

while True:
    pattern_choice = int(input("선택 (1-4): "))
    if 1 <= pattern_choice <= 4:
        break
    else:
        print("❌ 1~4 중에서 선택하세요!")

print(f"\n🎯 {size}줄 패턴을 '{char}' 문자로 출력합니다!")
print("="*40)

if pattern_choice == 1:
    # 이등변삼각형
    print("⭐ 커스텀 이등변삼각형")
    print("-" * 25)

    for i in range(1, size + 1):
        # 앞쪽 공백
        for j in range(size - i):
            print(" ", end="")

        # 선택한 문자 출력
        for j in range(2 * i - 1):
            print(char, end="")

        print()

elif pattern_choice == 2:
    # 다이아몬드
    print("💎 커스텀 다이아몬드")
    print("-" * 25)

    # 위쪽 삼각형
    for i in range(1, size + 1):
        for j in range(size - i):
            print(" ", end="")
        for j in range(2 * i - 1):
            print(char, end="")
        print()

    # 아래쪽 삼각형
    for i in range(size - 1, 0, -1):
        for j in range(size - i):
            print(" ", end="")
        for j in range(2 * i - 1):
            print(char, end="")
        print()

elif pattern_choice == 3:
    # 역삼각형
    print("🔻 커스텀 역삼각형")
    print("-" * 25)

    for i in range(size, 0, -1):
        for j in range(size - i):
            print(" ", end="")
        for j in range(2 * i - 1):
            print(char, end="")
        print()

elif pattern_choice == 4:
    # 속이 빈 삼각형
    print("🔺 커스텀 속이 빈 삼각형")
    print("-" * 25)

    for i in range(1, size + 1):
        for j in range(size - i):
            print(" ", end="")

        for j in range(2 * i - 1):
            if i == 1 or i == size:
                print(char, end="")
            else:
                if j == 0 or j == 2 * i - 2:
                    print(char, end="")
                else:
                    print(" ", end="")
        print()

print("\n🎉 커스텀 패턴 출력 완료!")
```

## 🔥 추가 기능 3: 점진적 패턴 시스템

```python
# ⭐ 점진적 패턴 시스템 (기본 변수와 반복문만 사용)
print("⭐ 점진적 패턴 출력 시스템")
print("="*35)

# 패턴 크기 입력
while True:
    height = int(input("패턴 크기 (3-12): "))
    if 3 <= height <= 12:
        break
    else:
        print("❌ 3~12 사이의 숫자를 입력하세요!")

# 사용할 문자 선택
print("\n🎨 사용할 문자를 선택하세요:")
print("1. 별 (*)")
print("2. 숫자 (1,2,3...)")
print("3. 알파벳 (A,B,C...)")
print("4. 특수문자 (♥)")

while True:
    char_choice = int(input("선택 (1-4): "))
    if 1 <= char_choice <= 4:
        break
    else:
        print("❌ 1~4 중에서 선택하세요!")

print(f"\n🎯 {height}줄 패턴을 한 줄씩 그려보겠습니다!")
print("="*50)

# 이등변삼각형을 한 줄씩 그리기
print("⭐ 이등변삼각형 - 단계별 출력")
print("-" * 30)

for i in range(1, height + 1):
    print(f"📏 {i}번째 줄:")

    # 앞쪽 공백
    for j in range(height - i):
        print(" ", end="")

    # 선택한 문자 출력
    for j in range(2 * i - 1):
        if char_choice == 1:  # 별
            print("*", end="")
        elif char_choice == 2:  # 숫자
            print(str(i), end="")
        elif char_choice == 3:  # 알파벳
            # A=65, B=66, C=67... (ASCII 코드)
            alphabet_index = (i - 1) % 26
            char = chr(65 + alphabet_index)  # A부터 시작
            print(char, end="")
        else:  # 특수문자
            print("♥", end="")

    print()  # 줄바꿈

    # 사용자 확인
    if i < height:
        continue_choice = input("다음 줄을 보시겠습니까? (엔터 또는 y): ")
        if continue_choice.lower() == 'q':
            break

    print()

print("\n🎉 점진적 패턴 출력 완료!")

# 패턴 변환 시스템
print("\n🔄 패턴 변환 시스템")
print("="*30)

print("같은 모양을 다른 문자로 변환해보겠습니다!")
print(f"📐 {height}줄 패턴을 네 가지 문자로 비교:")

# 네 가지 문자로 동시 출력
for i in range(1, height + 1):
    # 별 패턴
    for j in range(height - i):
        print(" ", end="")
    for j in range(2 * i - 1):
        print("*", end="")

    print("  ", end="")  # 패턴 간 간격

    # 숫자 패턴
    for j in range(height - i):
        print(" ", end="")
    for j in range(2 * i - 1):
        print(str(i), end="")

    print("  ", end="")  # 패턴 간 간격

    # 알파벳 패턴
    for j in range(height - i):
        print(" ", end="")
    for j in range(2 * i - 1):
        alphabet_index = (i - 1) % 26
        char = chr(65 + alphabet_index)
        print(char, end="")

    print("  ", end="")  # 패턴 간 간격

    # 특수문자 패턴
    for j in range(height - i):
        print(" ", end="")
    for j in range(2 * i - 1):
        print("♥", end="")

    print()  # 줄바꿈

print("\n🎨 패턴 변환 완료!")

# 크기 비교 시스템
print("\n📏 크기 비교 시스템")
print("="*30)

print("같은 패턴을 다른 크기로 비교해보겠습니다!")

# 3가지 크기로 비교 (3, 5, 7줄)
sizes = [3, 5, 7]
size_names = ["작은", "중간", "큰"]

for size_idx in range(3):
    current_size = sizes[size_idx]
    size_name = size_names[size_idx]

    print(f"\n⭐ {size_name} 크기 ({current_size}줄):")

    for i in range(1, current_size + 1):
        for j in range(current_size - i):
            print(" ", end="")
        for j in range(2 * i - 1):
            print("*", end="")
        print()

print("\n🎉 크기 비교 완료!")

# 진행률 표시 시스템
print("\n📊 진행률 표시 시스템")
print("="*30)

print("패턴 그리기 진행률을 표시합니다!")

for i in range(1, height + 1):
    # 진행률 계산 (기본 변수만 사용)
    progress = (i * 100) // height

    print(f"📊 진행률: {progress}% ({i}/{height})")

    # 진행률 바 표시
    print("진행률 바: [", end="")
    for j in range(10):
        if j < (progress // 10):
            print("█", end="")
        else:
            print("-", end="")
    print("]")

    # 현재 줄 출력
    print(f"현재 줄: ", end="")
    for j in range(height - i):
        print(" ", end="")
    for j in range(2 * i - 1):
        print("*", end="")
    print()

    print("-" * 30)

    # 잠시 대기 (사용자 확인)
    if i < height:
        input("다음 단계로 진행하려면 엔터를 누르세요...")

print("\n🎊 점진적 패턴 시스템 완료!")
print("🏆 모든 패턴을 성공적으로 그렸습니다!")
```

## 🎮 종합 메뉴 시스템

```python
# ⭐ 별찍기 아트 - 종합 메뉴 시스템

print("⭐ 별찍기 아트 마스터에 오신 것을 환영합니다!")
print("="*50)

while True:
    print("\n🎨 메뉴를 선택하세요:")
    print("="*30)
    print("1. 기본 패턴 (3가지)")
    print("2. 고급 패턴 (4가지)")
    print("3. 커스터마이징 시스템")
    print("4. 점진적 패턴 시스템")
    print("5. 종료")
    print("="*30)

    choice = input("선택하세요 (1-5): ")

    if choice == "1":
        # 기본 패턴
        print("\n⭐ 기본 패턴 출력")

        while True:
            height = int(input("패턴 크기 (3-15): "))
            if 3 <= height <= 15:
                break
            else:
                print("❌ 3~15 사이의 숫자를 입력하세요!")

        # 왼쪽 정렬 삼각형
        print("\n⭐ 왼쪽 정렬 삼각형")
        for i in range(1, height + 1):
            for j in range(i):
                print("*", end="")
            print()

        # 오른쪽 정렬 삼각형
        print("\n⭐ 오른쪽 정렬 삼각형")
        for i in range(1, height + 1):
            for j in range(height - i):
                print(" ", end="")
            for j in range(i):
                print("*", end="")
            print()

        # 이등변삼각형
        print("\n⭐ 이등변삼각형")
        for i in range(1, height + 1):
            for j in range(height - i):
                print(" ", end="")
            for j in range(2 * i - 1):
                print("*", end="")
            print()

    elif choice == "2":
        # 고급 패턴
        print("\n⭐ 고급 패턴 출력")

        while True:
            height = int(input("패턴 크기 (3-15): "))
            if 3 <= height <= 15:
                break
            else:
                print("❌ 3~15 사이의 숫자를 입력하세요!")

        print("\n💎 다이아몬드 패턴")
        # 위쪽 삼각형
        for i in range(1, height + 1):
            for j in range(height - i):
                print(" ", end="")
            for j in range(2 * i - 1):
                print("*", end="")
            print()
        # 아래쪽 삼각형
        for i in range(height - 1, 0, -1):
            for j in range(height - i):
                print(" ", end="")
            for j in range(2 * i - 1):
                print("*", end="")
            print()

    elif choice == "3":
        # 커스터마이징 시스템
        print("\n⭐ 커스터마이징 시스템")
        print("여기서 커스터마이징 코드 실행...")

    elif choice == "4":
        # 점진적 패턴 시스템
        print("\n⭐ 점진적 패턴 시스템")
        print("여기서 점진적 패턴 코드 실행...")

    elif choice == "5":
        print("\n👋 별찍기 아트를 이용해주셔서 감사합니다!")
        break

    else:
        print("❌ 1~5 중에서 선택하세요!")

print("🎉 프로그램을 종료합니다!")
```
