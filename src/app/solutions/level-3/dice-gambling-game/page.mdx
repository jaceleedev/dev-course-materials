# 🎲 주사위 도박 게임 (답안)

## 🚀 기본 기능 답안

```python
# 🎲 주사위 도박 게임 - 기본 기능 (기본 변수와 반복문만 사용)
print("🎲 주사위 카지노에 오신 것을 환영합니다!")
print("="*50)

# 시작 포인트
points = 1000
game_count = 0
print(f"💰 시작 포인트: {points}점")

while points > 0:
    print(f"\n💰 현재 포인트: {points}점")
    print("🎲 주사위 두 개의 합을 맞춰보세요! (2-12)")

    # 게임 계속 여부 확인
    continue_game = input("게임을 계속하시겠습니까? (y/n): ")
    if continue_game.lower() != 'y':
        break

    # 베팅 금액 입력
    while True:
        bet = int(input(f"베팅 금액 (10~{points}): "))
        if 10 <= bet <= points:
            break
        else:
            print(f"❌ 10~{points} 사이의 금액을 베팅하세요!")

    # 예측 입력
    while True:
        prediction = int(input("예측하는 합 (2-12): "))
        if 2 <= prediction <= 12:
            break
        else:
            print("❌ 2~12 사이의 숫자를 입력하세요!")

    # 주사위 굴리기 (간단한 의사 랜덤)
    game_count += 1
    dice1 = (game_count * 7) % 6 + 1
    dice2 = (game_count * 11) % 6 + 1
    total = dice1 + dice2

    print(f"\n🎲 주사위 결과: [{dice1}] [{dice2}] = {total}")

    # 결과 판정
    if total == prediction:
        winnings = bet * 5
        points += winnings
        print(f"🎉 정답! {bet} × 5 = {winnings}포인트 획득!")
    else:
        points -= bet
        print(f"😢 틀렸습니다. {bet}포인트 잃었습니다.")

    # 파산 경고
    if points <= 100:
        print("⚠️ 포인트가 부족합니다! 신중하게 베팅하세요!")

# 게임 종료
print(f"\n🎮 게임 종료!")
print(f"💰 최종 포인트: {points}점")

if points > 1000:
    print("🎉 축하합니다! 수익을 얻었습니다!")
elif points == 1000:
    print("😊 본전을 지켰습니다!")
else:
    print("😢 손실이 발생했습니다. 다음에 더 신중하게 해보세요!")
```

## 🎨 추가 기능 1: 고급 베팅 시스템

```python
# 🎲 주사위 도박 게임 - 고급 베팅 (기본 변수와 반복문만 사용)
print("🎲 고급 베팅 시스템!")
print("="*40)

points = 1000
game_count = 0
consecutive_wins = 0

print(f"💰 시작 포인트: {points}점")

while points > 0:
    print(f"\n💰 현재 포인트: {points}점")
    print(f"🔥 연속 승리: {consecutive_wins}회")

    print("\n🎯 베팅 종류를 선택하세요:")
    print("1. 정확한 숫자 맞추기 (5배)")
    print("2. 홀/짝 베팅 (1.8배)")
    print("3. 범위 베팅 (2~3배)")
    print("4. 럭키 세븐 (7배)")

    continue_game = input("게임을 계속하시겠습니까? (y/n): ")
    if continue_game.lower() != 'y':
        break

    # 베팅 종류 선택
    while True:
        bet_type = int(input("베팅 종류 선택 (1-4): "))
        if 1 <= bet_type <= 4:
            break
        else:
            print("❌ 1~4 중에서 선택하세요!")

    # 베팅 금액 입력
    while True:
        bet = int(input(f"베팅 금액 (10~{points}): "))
        if 10 <= bet <= points:
            break
        else:
            print(f"❌ 10~{points} 사이의 금액을 베팅하세요!")

    # 베팅 내용 입력
    if bet_type == 1:  # 정확한 숫자
        while True:
            prediction = int(input("예측하는 합 (2-12): "))
            if 2 <= prediction <= 12:
                break
            else:
                print("❌ 2~12 사이의 숫자를 입력하세요!")

    elif bet_type == 2:  # 홀/짝
        while True:
            odd_even = input("홀수(odd) 또는 짝수(even): ").lower()
            if odd_even == 'odd' or odd_even == 'even':
                break
            else:
                print("❌ 'odd' 또는 'even'을 입력하세요!")

    elif bet_type == 3:  # 범위
        print("1. 낮음 (2-6) - 3배")
        print("2. 중간 (7-8) - 4배")
        print("3. 높음 (9-12) - 3배")
        while True:
            range_choice = int(input("범위 선택 (1-3): "))
            if 1 <= range_choice <= 3:
                break
            else:
                print("❌ 1~3 중에서 선택하세요!")

    elif bet_type == 4:  # 럭키 세븐
        print("합이 7이 나오면 승리! (7배)")

    # 주사위 굴리기
    game_count += 1
    dice1 = (game_count * 7 + points) % 6 + 1
    dice2 = (game_count * 11 + consecutive_wins) % 6 + 1
    total = dice1 + dice2

    print(f"\n🎲 주사위 결과: [{dice1}] [{dice2}] = {total}")

    # 결과 판정
    win = False
    multiplier = 1

    if bet_type == 1:  # 정확한 숫자
        if total == prediction:
            win = True
            multiplier = 5
            print(f"🎉 정확한 숫자 적중!")

    elif bet_type == 2:  # 홀/짝
        is_odd = total % 2 == 1
        if (odd_even == 'odd' and is_odd) or (odd_even == 'even' and not is_odd):
            win = True
            multiplier = 1.8
            result_type = "홀수" if is_odd else "짝수"
            print(f"🎉 {result_type} 적중!")

    elif bet_type == 3:  # 범위
        if range_choice == 1 and 2 <= total <= 6:
            win = True
            multiplier = 3
            print("🎉 낮음 적중!")
        elif range_choice == 2 and 7 <= total <= 8:
            win = True
            multiplier = 4
            print("🎉 중간 적중!")
        elif range_choice == 3 and 9 <= total <= 12:
            win = True
            multiplier = 3
            print("🎉 높음 적중!")

    elif bet_type == 4:  # 럭키 세븐
        if total == 7:
            win = True
            multiplier = 7
            print("🎉 럭키 세븐! 대박!")

    # 점수 계산
    if win:
        winnings = int(bet * multiplier)
        points += winnings
        consecutive_wins += 1

        print(f"💰 획득: {bet} × {multiplier} = {winnings}포인트")

        # 연속 승리 보너스
        if consecutive_wins >= 3:
            bonus = consecutive_wins * 10
            points += bonus
            print(f"🔥 연속 승리 보너스! +{bonus}포인트")

    else:
        points -= bet
        consecutive_wins = 0
        print(f"😢 실패! {bet}포인트 잃었습니다.")

    # 파산 경고
    if points <= 100:
        print("⚠️ 포인트가 부족합니다!")

print(f"\n🎮 게임 종료! 최종 포인트: {points}점")
```

## 💡 추가 기능 2: 다양한 게임 모드

```python
# 🎲 주사위 도박 게임 - 다양한 게임 모드 (기본 변수와 반복문만 사용)
print("🎲 다양한 게임 모드!")
print("="*40)

points = 1000
game_count = 0
print(f"💰 시작 포인트: {points}점")

while points > 0:
    print(f"\n💰 현재 포인트: {points}점")
    print("\n🎮 게임 모드를 선택하세요:")
    print("1. 더블 오어 낫싱 (2배 또는 전부 잃기)")
    print("2. 잭팟 모드 (낮은 확률, 높은 배당)")
    print("3. 안전 모드 (높은 확률, 낮은 배당)")
    print("4. 종료")

    mode_choice = input("모드 선택 (1-4): ")

    if mode_choice == "4":
        break

    if mode_choice not in ["1", "2", "3"]:
        print("❌ 1~4 중에서 선택하세요!")
        continue

    # 베팅 금액 입력
    while True:
        bet = int(input(f"베팅 금액 (10~{points}): "))
        if 10 <= bet <= points:
            break
        else:
            print(f"❌ 10~{points} 사이의 금액을 베팅하세요!")

    # 주사위 굴리기
    game_count += 1
    dice1 = (game_count * 13 + points) % 6 + 1
    dice2 = (game_count * 17 + bet) % 6 + 1
    total = dice1 + dice2

    print(f"\n🎲 주사위 결과: [{dice1}] [{dice2}] = {total}")

    if mode_choice == "1":  # 더블 오어 낫싱
        prediction = int(input("예측하는 합 (2-12): "))

        if total == prediction:
            winnings = bet * 2
            points += winnings
            print(f"🎉 성공! 더블 획득! +{winnings}포인트")
        else:
            points = 0  # 전부 잃기
            print("😱 실패! 모든 포인트를 잃었습니다!")

    elif mode_choice == "2":  # 잭팟 모드
        print("🎰 잭팟 모드: 합이 7이면 50배!")

        if total == 7:
            winnings = bet * 50
            points += winnings
            print(f"🎊 잭팟! +{winnings}포인트")
        else:
            points -= bet
            print(f"😢 실패! {bet}포인트 잃음")

    elif mode_choice == "3":  # 안전 모드
        print("🛡️ 안전 모드: 합이 6~8이면 1.2배")

        if 6 <= total <= 8:
            winnings = int(bet * 1.2)
            points += winnings
            print(f"😊 안전하게 성공! +{winnings}포인트")
        else:
            points -= bet
            print(f"😢 실패! {bet}포인트 잃음")

    # 파산 확인
    if points <= 0:
        print("💸 파산! 게임 종료!")
        break

print(f"\n🎮 게임 종료! 최종 포인트: {points}점")
```
