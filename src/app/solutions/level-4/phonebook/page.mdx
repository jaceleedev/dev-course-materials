# 📞 스마트 전화번호부 (답안)

## 🚀 기본 기능 답안

```python
# 📞 스마트 전화번호부 - 기본 기능
print("📞 스마트 전화번호부에 오신 것을 환영합니다!")
print("="*50)

# 전체 연락처를 저장할 리스트
contacts = []

while True:
    print("\n📱 메뉴를 선택하세요:")
    print("="*25)
    print("1. 연락처 추가")
    print("2. 전체 연락처 보기")
    print("3. 연락처 삭제")
    print("4. 종료")
    print("="*25)

    choice = input("선택하세요 (1-4): ")

    if choice == "1":
        # 연락처 추가
        print("\n📱 새 연락처 추가")
        print("-"*20)

        name = input("이름: ")
        phone = input("전화번호: ")
        email = input("이메일: ")

        # 딕셔너리로 연락처 정보 저장
        new_contact = {
            "name": name,
            "phone": phone,
            "email": email
        }

        contacts.append(new_contact)
        print(f"✅ {name}님의 연락처가 추가되었습니다!")

    elif choice == "2":
        # 전체 연락처 보기
        print("\n📋 전체 연락처 목록")
        print("="*30)

        if len(contacts) == 0:
            print("저장된 연락처가 없습니다.")
        else:
            for i in range(len(contacts)):
                contact = contacts[i]
                print(f"{i+1}. 📞 {contact['name']}")
                print(f"   전화: {contact['phone']}")
                print(f"   이메일: {contact['email']}")
                print("-"*25)

    elif choice == "3":
        # 연락처 삭제
        if len(contacts) == 0:
            print("❌ 삭제할 연락처가 없습니다.")
        else:
            print("\n🗑️ 연락처 삭제")
            delete_name = input("삭제할 연락처 이름: ")

            found = False
            for i in range(len(contacts)):
                if contacts[i]["name"] == delete_name:
                    removed_contact = contacts[i]
                    del contacts[i]
                    print(f"✅ {removed_contact['name']}님의 연락처가 삭제되었습니다.")
                    found = True
                    break

            if not found:
                print("❌ 해당 이름의 연락처를 찾을 수 없습니다.")

    elif choice == "4":
        print("\n👋 전화번호부를 종료합니다!")
        break

    else:
        print("❌ 1~4 중에서 선택하세요!")

print("🎉 프로그램을 종료합니다!")
```
