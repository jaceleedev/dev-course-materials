# 📊 학점 계산기 (답안)

## 🚀 기본 기능 답안

```python
# 📊 학점 계산기
print("📊 학점 계산기에 오신 것을 환영합니다!")
print("="*40)

# 점수 입력 및 검증
score = int(input("점수를 입력하세요 (0-100): "))

if score < 0 or score > 100:
    print("❌ 잘못된 점수입니다. 0-100 사이의 값을 입력하세요.")
else:
    # 학점 계산
    if score >= 90:
        grade = "A"
        grade_point = 4.0
    elif score >= 80:
        grade = "B"
        grade_point = 3.0
    elif score >= 70:
        grade = "C"
        grade_point = 2.0
    elif score >= 60:
        grade = "D"
        grade_point = 1.0
    else:
        grade = "F"
        grade_point = 0.0

    # 결과 출력
    print("\n" + "="*40)
    print("📊 학점 계산 결과")
    print("="*40)
    print(f"📝 입력 점수: {score}점")
    print(f"🎓 학점: {grade}")
    print(f"📈 GPA: {grade_point}")
    print("="*40)
```

## 🎨 추가 기능 1: 장학금 대상 판정

```python
# 📊 학점 계산기 (장학금 판정 포함)
print("📊 학점 계산기 - 장학금 판정 시스템")
print("="*50)

# 기본 정보 입력
score = int(input("이번 학기 점수 (0-100): "))

if score < 0 or score > 100:
    print("❌ 잘못된 점수입니다.")
else:
    # 학점 계산
    if score >= 90:
        grade = "A"
        grade_point = 4.0
    elif score >= 80:
        grade = "B"
        grade_point = 3.0
    elif score >= 70:
        grade = "C"
        grade_point = 2.0
    elif score >= 60:
        grade = "D"
        grade_point = 1.0
    else:
        grade = "F"
        grade_point = 0.0

    # 장학금 판정을 위한 추가 정보
    previous_score = int(input("이전 학기 점수 (0-100): "))
    is_worker = input("근로학생인가요? (y/n): ")

    # 기본 결과 출력
    print("\n" + "="*50)
    print("📊 학점 계산 결과")
    print("="*50)
    print(f"📝 이번 학기 점수: {score}점")
    print(f"🎓 학점: {grade}")
    print(f"📈 GPA: {grade_point}")

    # 장학금 판정
    print("\n🎖️ 장학금 대상 여부")
    print("-"*30)

    scholarship_count = 0

    # 성적 우수 장학금
    if grade == "A":
        print("✅ 성적 우수 장학금 대상입니다!")
        scholarship_count += 1

    # 성적 향상 장학금
    if grade in ["A", "B"] and score > previous_score:
        print("✅ 성적 향상 장학금 대상입니다!")
        scholarship_count += 1

    # 근로 장학금
    if score >= 70 and is_worker == "y":
        print("✅ 근로 장학금 대상입니다!")
        scholarship_count += 1

    if scholarship_count == 0:
        print("❌ 해당하는 장학금이 없습니다.")
    else:
        print(f"\n🎉 총 {scholarship_count}개의 장학금 대상입니다!")

    print("="*50)
```

## 💡 추가 기능 2: 다중 과목 학점 계산

```python
# 📊 학점 계산기 (다중 과목)
print("📊 다중 과목 학점 계산기")
print("="*40)

# 3과목 점수 입력
print("3과목의 점수를 입력해주세요:")
subject1_score = int(input("과목 1 점수: "))
subject2_score = int(input("과목 2 점수: "))
subject3_score = int(input("과목 3 점수: "))

# 각 과목 학점 계산 함수 (조건문으로 대체)
def calculate_grade(score):
    if score >= 90:
        return "A", 4.0
    elif score >= 80:
        return "B", 3.0
    elif score >= 70:
        return "C", 2.0
    elif score >= 60:
        return "D", 1.0
    else:
        return "F", 0.0

# 각 과목별 학점 계산 (함수 대신 조건문 사용)
# 과목 1
if subject1_score >= 90:
    grade1, gpa1 = "A", 4.0
elif subject1_score >= 80:
    grade1, gpa1 = "B", 3.0
elif subject1_score >= 70:
    grade1, gpa1 = "C", 2.0
elif subject1_score >= 60:
    grade1, gpa1 = "D", 1.0
else:
    grade1, gpa1 = "F", 0.0

# 과목 2
if subject2_score >= 90:
    grade2, gpa2 = "A", 4.0
elif subject2_score >= 80:
    grade2, gpa2 = "B", 3.0
elif subject2_score >= 70:
    grade2, gpa2 = "C", 2.0
elif subject2_score >= 60:
    grade2, gpa2 = "D", 1.0
else:
    grade2, gpa2 = "F", 0.0

# 과목 3
if subject3_score >= 90:
    grade3, gpa3 = "A", 4.0
elif subject3_score >= 80:
    grade3, gpa3 = "B", 3.0
elif subject3_score >= 70:
    grade3, gpa3 = "C", 2.0
elif subject3_score >= 60:
    grade3, gpa3 = "D", 1.0
else:
    grade3, gpa3 = "F", 0.0

# 평균 계산
average_score = (subject1_score + subject2_score + subject3_score) / 3
average_gpa = (gpa1 + gpa2 + gpa3) / 3

# 평균 학점 계산
if average_score >= 90:
    average_grade = "A"
elif average_score >= 80:
    average_grade = "B"
elif average_score >= 70:
    average_grade = "C"
elif average_score >= 60:
    average_grade = "D"
else:
    average_grade = "F"

# 결과 출력
print("\n" + "="*40)
print("📊 다중 과목 학점 결과")
print("="*40)
print(f"과목 1: {subject1_score}점 → {grade1}학점 (GPA: {gpa1})")
print(f"과목 2: {subject2_score}점 → {grade2}학점 (GPA: {gpa2})")
print(f"과목 3: {subject3_score}점 → {grade3}학점 (GPA: {gpa3})")
print("-"*40)
print(f"📊 평균 점수: {average_score:.1f}점")
print(f"📊 평균 학점: {average_grade}")
print(f"📊 평균 GPA: {average_gpa:.2f}")
print("="*40)

# 성취도 평가
if average_gpa >= 3.5:
    print("🎉 우수한 성적입니다!")
elif average_gpa >= 2.5:
    print("👍 양호한 성적입니다!")
elif average_gpa >= 1.5:
    print("⚠️ 더 노력이 필요합니다!")
else:
    print("🚨 학습 상담이 필요합니다!")
```

## 💡 핵심 포인트

1. **조건문 활용**: if-elif-else를 사용한 구간별 학점 판정
2. **복합 조건**: and, or 연산자를 이용한 장학금 판정 로직
3. **입력 검증**: 잘못된 점수 범위에 대한 오류 처리
4. **사용자 경험**: 명확한 출력 형식과 시각적 구분선
5. **실용성**: 실제 성적 관리에 활용 가능한 기능들
