# 📊 학점 계산기 (답안)

## 🚀 기본 기능 답안

```python
# 📊 학점 계산기
print("📊 학점 계산기에 오신 것을 환영합니다!")
print("="*40)

# 점수 입력 및 검증
score = int(input("점수를 입력하세요 (0-100): "))

if score < 0 or score > 100:
    print("잘못된 점수입니다. 0-100 사이의 값을 입력하세요.")
else:
    # 학점 계산
    if score >= 90:
        grade = "A"
    elif score >= 80:
        grade = "B"
    elif score >= 70:
        grade = "C"
    elif score >= 60:
        grade = "D"
    else:
        grade = "F"

    # 결과 출력
    print("="*40)
    print("📊 학점 계산 결과")
    print("="*40)
    print(f"입력 점수: {score}점")
    print(f"학점: {grade}")
    print("="*40)
```

## 🎨 추가 기능 1: 장학금 판정 시스템

```python
# 📊 학점 계산기 (장학금 판정 포함)
print("📊 학점 계산기 - 장학금 판정 시스템")
print("="*50)

# 점수 입력 및 검증
score = int(input("점수를 입력하세요 (0-100): "))

if score < 0 or score > 100:
    print("잘못된 점수입니다. 0-100 사이의 값을 입력하세요.")
else:
    # 학점 계산
    if score >= 90:
        grade = "A"
    elif score >= 80:
        grade = "B"
    elif score >= 70:
        grade = "C"
    elif score >= 60:
        grade = "D"
    else:
        grade = "F"

    # 장학금 판정 로직
    if grade == "A":
        scholarship = "성적우수 장학금"
    elif grade == "B":
        scholarship = "성적향상 장학금"
    elif grade == "C":
        scholarship = "격려 장학금"
    else:
        scholarship = "해당없음"

    # 결과 출력
    print("="*50)
    print("📊 학점 계산 결과")
    print("="*50)
    print(f"입력 점수: {score}점")
    print(f"학점: {grade}")

    if scholarship != "해당없음":
        print(f"🎖️ 장학금: {scholarship} 대상입니다!")
    else:
        print("🎖️ 장학금: 해당하는 장학금이 없습니다.")

    print("="*50)
```

## 💡 추가 기능 2: 세부 학점 시스템

```python
# 📊 학점 계산기 (세부 학점 포함)
print("📊 학점 계산기 - 세부 학점 시스템")
print("="*50)

# 점수 입력 및 검증
score = int(input("점수를 입력하세요 (0-100): "))

if score < 0 or score > 100:
    print("잘못된 점수입니다. 0-100 사이의 값을 입력하세요.")

        grade = "B"

        grade = "C"
    elif score >= 60:
        grade = "D"

        grade = "F"

    # 세부 학점 계산
    if score >= 97:
        detailed_grade = "A+"
        next_grade = "최고 등급입니다!"

        next_grade = f"A+까지 {97 - score}점 부족"

        detailed_grade = "A-"
        next_grade = f"A0까지 {93 - score}점 부족"
    elif score >= 87:
        detailed_grade = "B+"
        next_grade = f"A-까지 {90 - score}점 부족"
    elif score >= 83:
        detailed_grade = "B0"
        next_grade = f"B+까지 {87 - score}점 부족"
    elif score >= 80:
        detailed_grade = "B-"

        detailed_grade = "C+"

        detailed_grade = "D+"

        detailed_grade = "D0"

    elif score >= 60:
        detailed_grade = "D-"
        next_grade = f"D0까지 {63 - score}점 부족"
    else:
        detailed_grade = "F"

        scholarship = "성적우수 장학금"

        scholarship = "성적향상 장학금"
    elif grade == "C":
        scholarship = "격려 장학금"

        scholarship = "해당없음"

    # 결과 출력
    print("="*50)
    print("📊 학점 계산 결과")
    print("="*50)
    print(f"입력 점수: {score}점")
    print(f"기본 학점: {grade}")
    print(f"세부 학점: {detailed_grade}")
    print(f"다음 등급: {next_grade}")

    if scholarship != "해당없음":
        print(f"🎖️ 장학금: {scholarship} 대상입니다!")
    else:
        print("🎖️ 장학금: 해당하는 장학금이 없습니다.")

    print("="*50)
```

## 🔥 도전 과제: 3과목 성적 관리

```python
# 📊 3과목 학점 계산기
print("📊 3과목 학점 계산기")
print("="*40)

# 3과목 점수 입력
print("3과목의 점수를 입력해주세요:")
math_score = int(input("수학 점수 (0-100): "))
english_score = int(input("영어 점수 (0-100): "))
science_score = int(input("과학 점수 (0-100): "))

# 수학 학점 계산
if math_score >= 90:
    math_grade = "A"
elif math_score >= 80:
    math_grade = "B"
elif math_score >= 70:
    math_grade = "C"
elif math_score >= 60:
    math_grade = "D"
else:
    math_grade = "F"

# 영어 학점 계산
if english_score >= 90:
    english_grade = "A"
elif english_score >= 80:
    english_grade = "B"
elif english_score >= 70:
    english_grade = "C"
elif english_score >= 60:
    english_grade = "D"
else:
    english_grade = "F"

# 과학 학점 계산
if science_score >= 90:
    science_grade = "A"
elif science_score >= 80:
    science_grade = "B"
elif science_score >= 70:
    science_grade = "C"
elif science_score >= 60:
    science_grade = "D"
else:
    science_grade = "F"

# 평균 계산
average_score = (math_score + english_score + science_score) / 3

# 평균 학점 계산
if average_score >= 90:
    average_grade = "A"
elif average_score >= 80:
    average_grade = "B"
elif average_score >= 70:
    average_grade = "C"
elif average_score >= 60:
    average_grade = "D"
else:
    average_grade = "F"

# 결과 출력
print("="*40)
print("📊 3과목 학점 결과")
print("="*40)
print(f"수학: {math_score}점 → {math_grade}학점")
print(f"영어: {english_score}점 → {english_grade}학점")
print(f"과학: {science_score}점 → {science_grade}학점")
print("-"*40)
print(f"📊 평균 점수: {average_score:.1f}점")
print(f"📊 평균 학점: {average_grade}")

# 성취도 평가
if average_score >= 90:
    print("🎉 우수한 성적입니다!")
elif average_score >= 80:
    print("👍 양호한 성적입니다!")
elif average_score >= 70:
    print("💪 조금 더 노력해보세요!")
elif average_score >= 60:
    print("⚠️ 더 많은 노력이 필요합니다!")
else:
    print("🚨 학습 상담이 필요합니다!")

print("="*40)
```

## 💡 핵심 포인트

1. **명확한 조건문**: if-elif-else로 점수 구간별 명확한 판정
2. **단계적 확장**: 기본 → 장학금 → 세부등급 순서로 기능 확장
3. **입력 검증**: 0-100 범위 벗어난 점수 처리
4. **사용자 친화적**: 명확한 출력 형식과 시각적 구분
5. **실용성**: 실제 성적 관리에 활용 가능한 기능들
