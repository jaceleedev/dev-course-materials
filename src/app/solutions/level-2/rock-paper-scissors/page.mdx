# ✂️ 가위바위보 게임 (답안)

## 🚀 기본 기능 답안

```python
import random

# ✂️ 가위바위보 게임
print("✂️📜🗿 가위바위보 게임에 오신 것을 환영합니다!")
print("="*50)

# 게임 규칙 설명
print("게임 규칙:")
print("✂️ 가위 (1) - 📜 보를 이김")
print("🗿 바위 (2) - ✂️ 가위를 이김")
print("📜 보 (3) - 🗿 바위를 이김")
print("="*50)

# 사용자 선택 입력
print("선택해주세요:")
print("1: 가위 ✂️")
print("2: 바위 🗿")
print("3: 보 📜")

user_choice = int(input("선택 (1, 2, 3): "))

# 입력 검증
if user_choice < 1 or user_choice > 3:
    print("❌ 잘못된 선택입니다! 1, 2, 3 중에서 선택해주세요.")
else:
    # 컴퓨터 랜덤 선택
    computer_choice = random.randint(1, 3)

    # 선택 변환
    choices = ["", "가위 ✂️", "바위 🗿", "보 📜"]

    print("\n" + "🎮" * 20)
    print("대결 결과")
    print("🎮" * 20)
    print(f"👤 당신의 선택: {choices[user_choice]}")
    print(f"🤖 컴퓨터 선택: {choices[computer_choice]}")
    print("-" * 30)

    # 승부 판정
    if user_choice == computer_choice:
        print("🤝 무승부입니다!")
        result = "무승부"
    elif (user_choice == 1 and computer_choice == 3) or \
         (user_choice == 2 and computer_choice == 1) or \
         (user_choice == 3 and computer_choice == 2):
        print("🎉 당신이 승리했습니다!")
        result = "승리"

        # 승리 이유 설명
        if user_choice == 1 and computer_choice == 3:
            print("✂️ 가위가 📜 보를 자릅니다!")
        elif user_choice == 2 and computer_choice == 1:
            print("🗿 바위가 ✂️ 가위를 부숩니다!")
        elif user_choice == 3 and computer_choice == 2:
            print("📜 보가 🗿 바위를 감쌉니다!")
    else:
        print("😢 컴퓨터가 승리했습니다!")
        result = "패배"

        # 패배 이유 설명
        if computer_choice == 1 and user_choice == 3:
            print("✂️ 가위가 📜 보를 자릅니다!")
        elif computer_choice == 2 and user_choice == 1:
            print("🗿 바위가 ✂️ 가위를 부숩니다!")
        elif computer_choice == 3 and user_choice == 2:
            print("📜 보가 🗿 바위를 감쌉니다!")

    print("🎮" * 20)
    print("게임을 즐겨주셔서 감사합니다!")
```

## 🎨 추가 기능 1: 점수 시스템과 연속 게임

```python
import random

# ✂️ 가위바위보 게임 (점수 시스템)
print("✂️📜🗿 가위바위보 대전!")
print("="*40)

# 점수 초기화
user_score = 0
computer_score = 0
game_count = 0

# 게임 진행
continue_game = "y"

while continue_game == "y":
    game_count += 1
    print(f"\n🎮 게임 {game_count}번째")
    print("="*40)

    # 현재 점수 표시
    print(f"📊 현재 점수 - 당신: {user_score} | 컴퓨터: {computer_score}")
    print("-" * 40)

    # 사용자 선택
    print("선택해주세요:")
    print("1: 가위 ✂️  2: 바위 🗿  3: 보 📜")

    user_choice = int(input("선택: "))

    # 입력 검증
    if user_choice < 1 or user_choice > 3:
        print("❌ 잘못된 선택입니다!")
        continue

    # 컴퓨터 선택
    computer_choice = random.randint(1, 3)

    # 선택 표시
    choices = ["", "가위 ✂️", "바위 🗿", "보 📜"]
    print(f"\n👤 당신: {choices[user_choice]}")
    print(f"🤖 컴퓨터: {choices[computer_choice]}")

    # 승부 판정 및 점수 업데이트
    if user_choice == computer_choice:
        print("🤝 무승부!")
    elif (user_choice == 1 and computer_choice == 3) or \
         (user_choice == 2 and computer_choice == 1) or \
         (user_choice == 3 and computer_choice == 2):
        print("🎉 당신 승리!")
        user_score += 1
    else:
        print("😢 컴퓨터 승리!")
        computer_score += 1

    # 계속 할지 묻기
    continue_game = input("\n다시 하시겠습니까? (y/n): ")

# 최종 결과
print("\n" + "🏆" * 30)
print("최종 결과")
print("🏆" * 30)
print(f"총 게임 수: {game_count}게임")
print(f"당신의 승수: {user_score}승")
print(f"컴퓨터 승수: {computer_score}승")
print(f"무승부: {game_count - user_score - computer_score}회")

# 최종 승자 발표
if user_score > computer_score:
    print("🎉 축하합니다! 당신이 최종 승리자입니다!")
elif computer_score > user_score:
    print("😢 아쉽게도 컴퓨터가 이겼습니다!")
else:
    print("🤝 최종 무승부입니다!")

# 승률 계산
if game_count > 0:
    win_rate = (user_score / game_count) * 100
    print(f"📈 당신의 승률: {win_rate:.1f}%")

print("🏆" * 30)
```

## 💡 추가 기능 2: 3판 2승제

```python
import random

# ✂️ 가위바위보 3판 2승제
print("✂️📜🗿 가위바위보 3판 2승제!")
print("="*50)
print("3판 중 2판을 먼저 이기는 사람이 승리합니다!")
print("="*50)

# 게임 변수 초기화
user_wins = 0
computer_wins = 0
round_results = []  # 각 라운드 결과 저장

# 1라운드
print("\n🏁 1라운드 시작!")
print("-" * 30)
print("1: 가위 ✂️  2: 바위 🗿  3: 보 📜")
user_choice1 = int(input("선택: "))
computer_choice1 = random.randint(1, 3)

choices = ["", "가위 ✂️", "바위 🗿", "보 📜"]
print(f"👤 당신: {choices[user_choice1]}")
print(f"🤖 컴퓨터: {choices[computer_choice1]}")

# 1라운드 결과 판정
if user_choice1 == computer_choice1:
    round1_result = "무승부"
    print("🤝 1라운드 무승부!")
elif (user_choice1 == 1 and computer_choice1 == 3) or \
     (user_choice1 == 2 and computer_choice1 == 1) or \
     (user_choice1 == 3 and computer_choice1 == 2):
    round1_result = "사용자 승리"
    user_wins += 1
    print("🎉 1라운드 당신 승리!")
else:
    round1_result = "컴퓨터 승리"
    computer_wins += 1
    print("😢 1라운드 컴퓨터 승리!")

round_results.append(f"1라운드: {round1_result}")

# 2라운드 (승부가 결정되지 않았다면)
if user_wins < 2 and computer_wins < 2:
    print("\n🏁 2라운드 시작!")
    print("-" * 30)
    print(f"현재 스코어 - 당신: {user_wins} | 컴퓨터: {computer_wins}")
    print("1: 가위 ✂️  2: 바위 🗿  3: 보 📜")
    user_choice2 = int(input("선택: "))
    computer_choice2 = random.randint(1, 3)

    print(f"👤 당신: {choices[user_choice2]}")
    print(f"🤖 컴퓨터: {choices[computer_choice2]}")

    # 2라운드 결과 판정
    if user_choice2 == computer_choice2:
        round2_result = "무승부"
        print("🤝 2라운드 무승부!")
    elif (user_choice2 == 1 and computer_choice2 == 3) or \
         (user_choice2 == 2 and computer_choice2 == 1) or \
         (user_choice2 == 3 and computer_choice2 == 2):
        round2_result = "사용자 승리"
        user_wins += 1
        print("🎉 2라운드 당신 승리!")
    else:
        round2_result = "컴퓨터 승리"
        computer_wins += 1
        print("😢 2라운드 컴퓨터 승리!")

    round_results.append(f"2라운드: {round2_result}")

# 3라운드 (아직 승부가 결정되지 않았다면)
if user_wins < 2 and computer_wins < 2:
    print("\n🏁 3라운드 시작! (최종 라운드)")
    print("-" * 30)
    print(f"현재 스코어 - 당신: {user_wins} | 컴퓨터: {computer_wins}")
    print("1: 가위 ✂️  2: 바위 🗿  3: 보 📜")
    user_choice3 = int(input("선택: "))
    computer_choice3 = random.randint(1, 3)

    print(f"👤 당신: {choices[user_choice3]}")
    print(f"🤖 컴퓨터: {choices[computer_choice3]}")

    # 3라운드 결과 판정
    if user_choice3 == computer_choice3:
        round3_result = "무승부"
        print("🤝 3라운드 무승부!")
    elif (user_choice3 == 1 and computer_choice3 == 3) or \
         (user_choice3 == 2 and computer_choice3 == 1) or \
         (user_choice3 == 3 and computer_choice3 == 2):
        round3_result = "사용자 승리"
        user_wins += 1
        print("🎉 3라운드 당신 승리!")
    else:
        round3_result = "컴퓨터 승리"
        computer_wins += 1
        print("😢 3라운드 컴퓨터 승리!")

    round_results.append(f"3라운드: {round3_result}")

# 최종 결과 발표
print("\n" + "🏆" * 40)
print("3판 2승제 최종 결과")
print("🏆" * 40)

# 각 라운드 결과 표시
print("📋 라운드별 결과:")
for result in round_results:
    print(f"  {result}")

print(f"\n📊 최종 스코어")
print(f"👤 당신: {user_wins}승")
print(f"🤖 컴퓨터: {computer_wins}승")

# 최종 승자 결정
if user_wins > computer_wins:
    print("\n🎉🎉🎉 축하합니다! 당신이 3판 2승제의 최종 승리자입니다! 🎉🎉🎉")
    print("🏆 우승 트로피를 드립니다!")
elif computer_wins > user_wins:
    print("\n😢 아쉽네요. 컴퓨터가 3판 2승제의 승리자입니다.")
    print("💪 다음엔 더 좋은 결과가 있을 거예요!")
else:
    print("\n🤝 놀랍게도 최종 무승부입니다!")
    print("👏 박빙의 승부였어요!")

print("🏆" * 40)
print("게임을 즐겨주셔서 감사합니다!")
```

## 💡 핵심 포인트

1. **랜덤 선택**: `random.randint(1, 3)`로 컴퓨터의 무작위 선택 구현
2. **완전한 승부 로직**: 모든 경우의 수를 조건문으로 처리
3. **입력 검증**: 잘못된 사용자 입력에 대한 적절한 처리
4. **점수 시스템**: 연속 게임과 누적 점수 관리
5. **3판 2승제**: 반복문 없이 개별 라운드를 조건문으로 구현
6. **사용자 경험**: 시각적 구분선과 명확한 결과 표시
