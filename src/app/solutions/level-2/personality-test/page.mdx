# 🧠 미니 심리테스트 (답안)

## 🚀 기본 기능 답안

```python
# 🧠 미니 심리테스트
print("🧠 성격 유형 테스트에 오신 것을 환영합니다!")
print("="*50)
print("3가지 질문에 답하여 당신의 성격 유형을 알아보세요!")
print("="*50)

score = 0

# Q1: 사회성 테스트
print("\nQ1. 새로운 환경에서 당신은?")
print("1. 먼저 다가가서 인사한다 ✋")
print("2. 상황을 보고 적절히 행동한다 👀")
print("3. 누군가 다가올 때까지 기다린다 🤐")

choice1 = int(input("선택 (1, 2, 3): "))

if choice1 == 1:
    score += 3
elif choice1 == 2:
    score += 2
else:
    score += 1

# Q2: 도전성 테스트
print("\nQ2. 새로운 취미를 시작할 때?")
print("1. 바로 도전해본다 🚀")
print("2. 조금 알아보고 시작한다 🔍")
print("3. 충분히 준비한 후 시작한다 📚")

choice2 = int(input("선택 (1, 2, 3): "))

if choice2 == 1:
    score += 3
elif choice2 == 2:
    score += 2
else:
    score += 1

# Q3: 의사결정 스타일
print("\nQ3. 중요한 결정을 할 때?")
print("1. 직감을 믿고 결정한다 💫")
print("2. 장단점을 비교해본다 ⚖️")
print("3. 많은 정보를 수집한 후 결정한다 📊")

choice3 = int(input("선택 (1, 2, 3): "))

if choice3 == 1:
    score += 3
elif choice3 == 2:
    score += 2
else:
    score += 1

# 성격 유형 판정
print("\n" + "🎭" * 25)
print("     심리테스트 결과")
print("🎭" * 25)

if score >= 7:
    personality_type = "모험가형 🏃‍♂️"
    description = "새로운 경험을 좋아하고 활동적인 성격입니다."
    advice = "다양한 취미활동에 도전해보세요!"
    character = "🦁 사자 - 용감하고 리더십이 강해요!"
elif score >= 4:
    personality_type = "안정추구형 🏠"
    description = "균형잡힌 사고와 안정을 추구하는 성격입니다."
    advice = "꾸준한 계획을 세워 목표를 달성해보세요!"
    character = "🐻 곰 - 든든하고 신뢰할 수 있어요!"
else:
    personality_type = "신중형 🤔"
    description = "깊이 있게 생각하고 신중한 판단을 하는 성격입니다."
    advice = "충분한 계획을 세운 후 행동하는 것이 좋겠어요!"
    character = "🦉 올빼미 - 지혜롭고 사려 깊어요!"

print(f"✨ 성격 유형: {personality_type}")
print(f"📝 설명: {description}")
print(f"💡 조언: {advice}")
print(f"🐾 동물 캐릭터: {character}")
print(f"📊 점수: {score}/9")
print("🎭" * 25)
```

## 💡 추가 기능 2: 상세 분석 및 점수

```python
# 🧠 종합 심리테스트 (상세 분석)
print("🧠 종합 심리테스트 - 상세 분석")
print("="*50)

# 점수 저장 변수들
social_score = 0      # 사회성 점수
challenge_score = 0   # 도전성 점수
decision_score = 0    # 의사결정 점수
total_score = 0

print("5가지 영역에서 당신의 성향을 분석합니다!")
print("="*50)

# Q1: 사회성
print("\nQ1. 파티에서 당신은?")
print("1. 새로운 사람들과 적극적으로 대화한다 (3점)")
print("2. 아는 사람들과 이야기한다 (2점)")
print("3. 조용히 관찰한다 (1점)")

choice1 = int(input("선택: "))
if choice1 == 1:
    social_score += 3
elif choice1 == 2:
    social_score += 2
else:
    social_score += 1

# Q2: 도전성
print("\nQ2. 새로운 프로젝트를 맡았을 때?")
print("1. 즉시 시작해서 진행하며 배운다 (3점)")
print("2. 기본 계획을 세우고 시작한다 (2점)")
print("3. 철저히 준비한 후 시작한다 (1점)")

choice2 = int(input("선택: "))
if choice2 == 1:
    challenge_score += 3
elif choice2 == 2:
    challenge_score += 2
else:
    challenge_score += 1

# Q3: 의사결정
print("\nQ3. 어려운 선택을 해야 할 때?")
print("1. 직감을 따른다 (3점)")
print("2. 주변 의견을 듣는다 (2점)")
print("3. 데이터를 분석한다 (1점)")

choice3 = int(input("선택: "))
if choice3 == 1:
    decision_score += 3
elif choice3 == 2:
    decision_score += 2
else:
    decision_score += 1

total_score = social_score + challenge_score + decision_score

# 상세 분석 결과
print("\n" + "📊" * 30)
print("     상세 분석 결과")
print("📊" * 30)

# 사회성 분석
print(f"\n👥 사회성 점수: {social_score}/3")
if social_score == 3:
    social_type = "외향적 리더형"
    social_advice = "팀 프로젝트나 사교활동에 참여해보세요!"
elif social_score == 2:
    social_type = "적응형"
    social_advice = "상황에 맞게 잘 적응하는 능력을 활용하세요!"
else:
    social_type = "관찰자형"
    social_advice = "깊이 있는 인간관계를 만들어가세요!"

print(f"   유형: {social_type}")
print(f"   조언: {social_advice}")

# 도전성 분석
print(f"\n🚀 도전성 점수: {challenge_score}/3")
if challenge_score == 3:
    challenge_type = "모험가형"
    challenge_advice = "새로운 기술이나 취미에 도전해보세요!"
elif challenge_score == 2:
    challenge_type = "균형추구형"
    challenge_advice = "계획과 실행의 균형을 맞춰가세요!"
else:
    challenge_type = "신중파"
    challenge_advice = "체계적인 학습이나 연구에 집중해보세요!"

print(f"   유형: {challenge_type}")
print(f"   조언: {challenge_advice}")

# 의사결정 분석
print(f"\n🧠 의사결정 점수: {decision_score}/3")
if decision_score == 3:
    decision_type = "직관형"
    decision_advice = "빠른 판단력을 활용한 분야에 도전하세요!"
elif decision_score == 2:
    decision_type = "소통형"
    decision_advice = "팀워크를 중시하는 환경에서 능력을 발휘하세요!"
else:
    decision_type = "분석형"
    decision_advice = "데이터 분석이나 연구 분야를 고려해보세요!"

print(f"   유형: {decision_type}")
print(f"   조언: {decision_advice}")

# 종합 분석
print(f"\n🎯 종합 점수: {total_score}/9")
if total_score >= 8:
    overall_type = "리더십형"
    career_recommendation = "경영, 마케팅, 영업"
    celebrity = "스티브 잡스"
elif total_score >= 6:
    overall_type = "균형형"
    career_recommendation = "교육, 상담, 기획"
    celebrity = "오프라 윈프리"
else:
    overall_type = "전문가형"
    career_recommendation = "연구, 개발, 분석"
    celebrity = "알버트 아인슈타인"

print(f"\n🌟 종합 성격 유형: {overall_type}")
print(f"💼 추천 직업군: {career_recommendation}")
print(f"👑 닮은 유명인: {celebrity}")

# 백분율 표시
percentage = (total_score / 9) * 100
print(f"\n📈 활동성 지수: {percentage:.1f}%")

if percentage >= 80:
    activity_level = "매우 활발함"
elif percentage >= 60:
    activity_level = "활발함"
elif percentage >= 40:
    activity_level = "보통"
else:
    activity_level = "신중함"

print(f"🎭 활동성 수준: {activity_level}")
print("📊" * 30)
```

## 💡 핵심 포인트

1. **점수 시스템**: 각 선택에 따른 점수 누적으로 성격 유형 판정
2. **다양한 분석**: 여러 영역별 세부 분석 제공
3. **MBTI 구현**: 4개 차원을 개별 질문으로 처리
4. **개인화된 결과**: 성격 유형별 맞춤 조언과 추천사항
5. **재미있는 요소**: 동물 캐릭터, 유명인 매칭 등으로 흥미 증대
