# 🎯 간단한 성격 테스트 (답안)

## 🚀 기본 기능 답안

```python
# 🎯 간단한 성격 테스트
print("🎯 성격 테스트를 시작합니다!")
print("="*40)

# 질문 1: 사회성
print("Q1. 새로운 사람을 만나면?")
print("1. 먼저 말을 건다")
print("2. 상대방이 말을 걸 때까지 기다린다")
q1 = int(input("선택 (1 또는 2): "))

# 질문 2: 활동성
print("\nQ2. 주말에 하고 싶은 것은?")
print("1. 밖에 나가서 활동한다")
print("2. 집에서 쉰다")
q2 = int(input("선택 (1 또는 2): "))

# 질문 3: 계획성
print("\nQ3. 여행을 갈 때?")
print("1. 미리 계획을 세운다")
print("2. 즉흥적으로 간다")
q3 = int(input("선택 (1 또는 2): "))

# 8가지 경우의 수로 성격 유형 결정
if q1 == 1 and q2 == 1 and q3 == 1:
    personality = "계획적 리더형"
elif q1 == 1 and q2 == 1 and q3 == 2:
    personality = "자유로운 모험가형"
elif q1 == 1 and q2 == 2 and q3 == 1:
    personality = "차분한 계획형"
elif q1 == 1 and q2 == 2 and q3 == 2:
    personality = "편안한 소통형"
elif q1 == 2 and q2 == 1 and q3 == 1:
    personality = "신중한 활동형"
elif q1 == 2 and q2 == 1 and q3 == 2:
    personality = "즉흥적 탐험형"
elif q1 == 2 and q2 == 2 and q3 == 1:
    personality = "조용한 계획형"
else:  # q1 == 2 and q2 == 2 and q3 == 2
    personality = "자유로운 휴식형"

# 결과 출력
print("\n" + "🎭" * 25)
print("     성격 테스트 결과")
print("🎭" * 25)
print(f"✨ 당신의 성격 유형: {personality}")
print("🎭" * 25)
```

## 💡 핵심 포인트

1. **명확한 8가지 경우**: 3개 질문 × 2개 선택지로 모든 경우의 수 처리
2. **조건문 중심**: and 연산자로 복합 조건 처리
3. **단계적 확장**: 기본 → 상세 분석 → 동물 캐릭터 순서로 기능 확장
4. **개별 분석**: 각 영역(사회성, 활동성, 계획성)을 별도로 분석
5. **재미있는 요소**: 동물 캐릭터와 맞춤 조언으로 흥미 증대
