# 💼 나만의 명함 만들기 (답안)

## 🚀 기본 기능 답안

```python
# 💼 나만의 명함 만들기
print("💼 나만의 디지털 명함 만들기")
print("="*40)

# 개인 정보 입력
name = input("이름: ")
age = int(input("나이: "))
job = input("직업: ")
hobby = input("취미: ")
email = input("이메일: ")
phone = input("전화번호: ")

# 명함 출력
print("\n" + "="*40)
print("✨ 나만의 디지털 명함 ✨")
print("="*40)
print(f"👤 이름: {name}")
print(f"🎂 나이: {age}세")
print(f"💼 직업: {job}")
print(f"🎨 취미: {hobby}")
print(f"📧 이메일: {email}")
print(f"📱 전화번호: {phone}")
print("="*40)
print("감사합니다! 😊")
```

## 🎨 추가 기능 1: ASCII 아트 장식

```python
# 💼 나만의 명함 만들기 (ASCII 아트 버전)
print("💼 나만의 디지털 명함 만들기")
print("="*40)

# 개인 정보 입력
name = input("이름: ")
age = int(input("나이: "))
job = input("직업: ")
hobby = input("취미: ")
email = input("이메일: ")
phone = input("전화번호: ")

# ASCII 아트 장식된 명함 출력
print("\n")
print("┌─────────────────────────────────────┐")
print("│        ✨ 디지털 명함 ✨           │")
print("├─────────────────────────────────────┤")
print(f"│ 👤 이름: {name:<25} │")
print(f"│ 🎂 나이: {age}세{'':<22} │")
print(f"│ 💼 직업: {job:<25} │")
print(f"│ 🎨 취미: {hobby:<25} │")
print("├─────────────────────────────────────┤")
print(f"│ 📧 {email:<30} │")
print(f"│ 📱 {phone:<30} │")
print("└─────────────────────────────────────┘")
```

## 🏆 추가 기능 2: QR코드 스타일

```python
# 💼 나만의 명함 만들기 (QR코드 스타일)
print("💼 나만의 디지털 명함 만들기")
print("="*40)

# 기본 정보 입력
name = input("이름: ")
age = int(input("나이: "))
job = input("직업: ")
hobby = input("취미: ")
email = input("이메일: ")
phone = input("전화번호: ")

# 추가 정보 입력
favorite_color = input("좋아하는 색상: ")
motto = input("좌우명 또는 슬로건: ")

# QR코드 스타일 명함 출력
print("\n" + "="*50)
print("✨ 나만의 디지털 명함 ✨")
print("="*50)
print(f"👤 이름: {name}")
print(f"🎂 나이: {age}세")
print(f"💼 직업: {job}")
print(f"🎨 취미: {hobby}")
print(f"💫 좌우명: \"{motto}\"")
print(f"🌈 좋아하는 색상: {favorite_color}")
print("="*50)
print(f"📧 이메일: {email}")
print(f"📱 전화번호: {phone}")
print("="*50)

# 간단한 QR코드 스타일 (ASCII)
print("\n📱 연락처 QR코드 (디자인)")
print("█████████████████")
print("█  ██   ██  █████")
print("█  ██   ██  █████")
print("█  ██   ██  █████")
print("█████████████████")
print(f"☝️  {name}님과 연락하세요!")
```

## 💡 핵심 포인트

1. **변수 활용**: 사용자 입력을 변수에 저장하여 재사용
2. **문자열 포맷팅**: f-string을 사용하여 변수를 문자열에 삽입
3. **출력 디자인**: 적절한 구분선과 이모지로 시각적 효과 추가
4. **ASCII 아트**: 특수 문자를 활용한 테두리 디자인
