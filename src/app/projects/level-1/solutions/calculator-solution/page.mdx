# 🧮 계산기 (답안)

## 🚀 기본 기능 답안

```python
# 🧮 간단한 계산기
print("🧮 간단한 계산기")
print("="*30)

# 두 숫자 입력
num1 = float(input("첫 번째 숫자: "))
num2 = float(input("두 번째 숫자: "))

# 사칙연산 계산
addition = num1 + num2
subtraction = num1 - num2
multiplication = num1 * num2
division = num1 / num2

# 결과 출력
print("\n" + "="*30)
print("📊 계산 결과")
print("="*30)
print(f"{num1} + {num2} = {addition}")
print(f"{num1} - {num2} = {subtraction}")
print(f"{num1} × {num2} = {multiplication}")
print(f"{num1} ÷ {num2} = {division:.2f}")
```

## 🌍 추가 기능 1: 단위 변환기

```python
# 🧮 간단한 계산기 + 단위 변환기
print("🧮 간단한 계산기 + 단위 변환기")
print("="*40)

# 기본 사칙연산
num1 = float(input("첫 번째 숫자: "))
num2 = float(input("두 번째 숫자: "))

addition = num1 + num2
subtraction = num1 - num2
multiplication = num1 * num2
division = num1 / num2

print("\n" + "="*30)
print("📊 기본 계산 결과")
print("="*30)
print(f"{num1} + {num2} = {addition}")
print(f"{num1} - {num2} = {subtraction}")
print(f"{num1} × {num2} = {multiplication}")
print(f"{num1} ÷ {num2} = {division:.2f}")

# 단위 변환기
print("\n🌍 단위 변환기")
print("="*30)

# 거리 변환 (km ↔ mile)
km = float(input("킬로미터 입력: "))
mile_from_km = km * 0.621371
print(f"{km} km = {mile_from_km:.2f} mile")

mile = float(input("마일 입력: "))
km_from_mile = mile / 0.621371
print(f"{mile} mile = {km_from_mile:.2f} km")

# 온도 변환 (°C ↔ °F)
celsius = float(input("섭씨 온도 입력: "))
fahrenheit_from_celsius = (celsius * 9/5) + 32
print(f"{celsius}°C = {fahrenheit_from_celsius:.1f}°F")

fahrenheit = float(input("화씨 온도 입력: "))
celsius_from_fahrenheit = (fahrenheit - 32) * 5/9
print(f"{fahrenheit}°F = {celsius_from_fahrenheit:.1f}°C")
```

## 💰 추가 기능 2: 생활 계산기

```python
# 🧮 다기능 계산기 (생활 계산기 포함)
print("🧮 다기능 계산기")
print("="*40)

# 기본 사칙연산
print("📊 기본 계산기")
print("-"*20)
num1 = float(input("첫 번째 숫자: "))
num2 = float(input("두 번째 숫자: "))

print(f"{num1} + {num2} = {num1 + num2}")
print(f"{num1} - {num2} = {num1 - num2}")
print(f"{num1} × {num2} = {num1 * num2}")
print(f"{num1} ÷ {num2} = {num1 / num2:.2f}")

# 단위 변환
print("\n🌍 단위 변환기")
print("-"*20)
km = float(input("킬로미터: "))
print(f"{km} km = {km * 0.621371:.2f} mile")

celsius = float(input("섭씨 온도: "))
print(f"{celsius}°C = {(celsius * 9/5) + 32:.1f}°F")

# 생활 계산기
print("\n💰 생활 계산기")
print("-"*20)

# 할인 계산기
original_price = float(input("원래 가격: "))
discount_rate = float(input("할인율 (%): "))
discount_amount = original_price * (discount_rate / 100)
final_price = original_price - discount_amount

print(f"원래 가격: {original_price:,.0f}원")
print(f"할인 금액: {discount_amount:,.0f}원")
print(f"최종 가격: {final_price:,.0f}원")

# 팁 계산기
meal_cost = float(input("식사 비용: "))
tip_rate = float(input("팁 비율 (%): "))
tip_amount = meal_cost * (tip_rate / 100)
total_cost = meal_cost + tip_amount

print(f"식사 비용: {meal_cost:,.0f}원")
print(f"팁 금액: {tip_amount:,.0f}원")
print(f"총 비용: {total_cost:,.0f}원")
```

## 📐 추가 기능 3: 수학 계산기

```python
# 🧮 완전체 계산기 (수학 계산기 포함)
print("🧮 완전체 계산기")
print("="*40)

# 기본 사칙연산
print("📊 기본 계산기")
num1 = float(input("첫 번째 숫자: "))
num2 = float(input("두 번째 숫자: "))
print(f"결과: {num1} + {num2} = {num1 + num2}")
print(f"결과: {num1} × {num2} = {num1 * num2}")

# 단위 변환
print("\n🌍 단위 변환")
km = float(input("킬로미터: "))
print(f"변환: {km} km = {km * 0.621371:.2f} mile")

# 생활 계산
print("\n💰 할인 계산")
price = float(input("가격: "))
discount = float(input("할인율 (%): "))
final = price * (1 - discount/100)
print(f"할인가: {final:,.0f}원")

# 수학 계산기
print("\n📐 수학 계산기")
print("-"*20)

# 원의 계산
radius = float(input("원의 반지름: "))
pi = 3.14159
circle_area = pi * radius * radius
circle_perimeter = 2 * pi * radius

print(f"원의 넓이: {circle_area:.2f}")
print(f"원의 둘레: {circle_perimeter:.2f}")

# 직사각형 계산
length = float(input("직사각형 가로: "))
width = float(input("직사각형 세로: "))
rectangle_area = length * width
rectangle_perimeter = 2 * (length + width)

print(f"직사각형 넓이: {rectangle_area:.2f}")
print(f"직사각형 둘레: {rectangle_perimeter:.2f}")

# 삼각형 계산
base = float(input("삼각형 밑변: "))
height = float(input("삼각형 높이: "))
triangle_area = (base * height) / 2

print(f"삼각형 넓이: {triangle_area:.2f}")
```

## 💡 핵심 포인트

1. **변수 활용**: 계산 결과를 변수에 저장하여 재사용
2. **타입 변환**: `float()` 함수로 문자열을 숫자로 변환
3. **포맷팅**: 소수점 자리수 조절 (`.2f`) 및 천단위 콤마 (`:,`)
4. **수학 상수**: π 값을 변수로 정의하여 사용
